FROM Воспитатели

WITH CTE AS (
    SELECT ФИО, 
           ROW_NUMBER() OVER (ORDER BY ФИО) AS RowNum,
           COUNT(*) OVER () AS TotalCount
    FROM Воспитатели
)
SELECT ФИО
FROM CTE
WHERE RowNum <= TotalCount * 0.1;
